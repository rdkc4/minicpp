int main(){
    int x;
    int y;
    int z;
    int f;
    if(x>y)
        y = xyz(x,y,z) + x + y;
    else{
        x = z - x + f;
    }

    if(x>y)
        if(z>f)
            return x;
        else
            return x;
    else
        x = x - 1;

    return;
}

int xyz(int x, int y, int z){
    return x+(y-z);
}
